### 2. `services/image_extractor.py`
Extrai imagens de uma página web ou processa upload:

```python
import requests
from bs4 import BeautifulSoup
from typing import List
from urllib.parse import urljoin, urlparse

class ImageExtractor:
    @staticmethod
    def extract_from_url(page_url: str) -> List[str]:
        """Extrai todas as URLs de imagens de uma página"""
        try:
            headers = {
                'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
            }
            response = requests.get(page_url, headers=headers, timeout=10)
            soup = BeautifulSoup(response.content, 'html.parser')
            
            images = []
            for img in soup.find_all('img'):
                src = img.get('src') or img.get('data-src')
                if src:
                    full_url = urljoin(page_url, src)
                    # Filtrar apenas imagens válidas
                    if any(full_url.lower().endswith(ext) for ext in ['.jpg', '.jpeg', '.png', '.webp', '.gif']):
                        images.append(full_url)
            return images
        except Exception as e:
            raise Exception(f"Erro ao extrair imagens: {str(e)}")

    @staticmethod
    def validate_image_url(url: str) -> bool:
        """Valida se URL é uma imagem acessível"""
        try:
            headers = {'User-Agent': 'Mozilla/5.0'}
            response = requests.head(url, headers=headers, timeout=5)
            content_type = response.headers.get('content-type', '')
            return 'image' in content_type
        except:
            return False
```

### 3. `services/video_generator.py`
Gera o vídeo usando MoviePy:

```python
from moviepy.editor import ImageClip, concatenate_videoclips, AudioFileClip
from moviepy.video.fx.all import fadein, fadeout
import tempfile
import os
import requests
from typing import List, Optional
from PIL import Image
import io

class VideoGenerator:
    def __init__(self):
        self.temp_dir = tempfile.mkdtemp()
    
    def download_image(self, url: str, index: int) -> str:
        """Baixa imagem e retorna caminho local"""
        try:
            headers = {'User-Agent': 'Mozilla/5.0'}
            response = requests.get(url, headers=headers, timeout=10)
            img = Image.open(io.BytesIO(response.content))
            
            # Converter para RGB se necessário
            if img.mode in ('RGBA', 'P'):
                img = img.convert('RGB')
            
            path = os.path.join(self.temp_dir, f"img_{index:04d}.jpg")
            img.save(path, quality=95)
            return path
        except Exception as e:
            raise Exception(f"Erro ao baixar imagem {url}: {str(e)}")
    
    def create_video(
        self, 
        image_urls: List[str], 
        duration_per_image: float = 3.0,
        transition_duration: float = 0.5,
        resolution: tuple = (1920, 1080),
        fps: int = 24,
        add_fade: bool = True
    ) -> str:
        """
        Cria vídeo a partir de lista de URLs de imagens
        
        Args:
            image_urls: Lista de URLs das imagens
            duration_per_image: Duração de cada imagem em segundos
            transition_duration: Duração da transição em segundos
            resolution: Resolução do vídeo (largura, altura)
            fps: Frames por segundo
            add_fade: Adicionar efeito fade entre imagens
        """
        clips = []
        
        print(f"Processando {len(image_urls)} imagens...")
        
        for i, url in enumerate(image_urls):
            try:
                # Baixar imagem
                local_path = self.download_image(url, i)
                
                # Criar clip
                clip = ImageClip(local_path, duration=duration_per_image)
                
                # Redimensionar mantendo aspect ratio (letterbox se necessário)
                clip = clip.resize(height=resolution[1])
                if clip.w > resolution[0]:
                    clip = clip.resize(width=resolution[0])
                
                # Centralizar
                clip = clip.set_position('center')
                
                # Adicionar fade
                if add_fade and i > 0:
                    clip = fadein(clip, transition_duration)
                if add_fade and i < len(image_urls) - 1:
                    clip = fadeout(clip, transition_duration)
                
                clips.append(clip)
                print(f"✓ Processada imagem {i+1}/{len(image_urls)}")
                
            except Exception as e:
                print(f"✗ Erro na imagem {i+1}: {str(e)}")
                continue
        
        if not clips:
            raise Exception("Nenhuma imagem foi processada com sucesso")
        
        # Concatenar clips
        final_video = concatenate_videoclips(clips, method="compose")
        
        # Definir tamanho do canvas
        final_video = final_video.set_duration(sum([c.duration for c in clips]))
        
        # Salvar vídeo
        output_path = os.path.join(self.temp_dir, "output.mp4")
        final_video.write_videofile(
            output_path,
            fps=fps,
            codec='libx264',
            audio=False,
            threads=4,
            preset='medium',
            logger=None  # Remove logs verbosos do ffmpeg
        )
        
        # Limpar imagens temporárias
        for i in range(len(image_urls)):
            img_path = os.path.join(self.temp_dir, f"img_{i:04d}.jpg")
            if os.path.exists(img_path):
                os.remove(img_path)
        
        return output_path
```

from fastapi import FastAPI, HTTPException, BackgroundTasks, UploadFile, File
from fastapi.responses import FileResponse, StreamingResponse
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel, HttpUrl
from typing import List, Optional
import os
import shutil
import uuid

from services.image_extractor import ImageExtractor
from services.video_generator import VideoGenerator

app = FastAPI(title="Video Creator API", version="1.0")

# CORS para permitir acesso do frontend
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# Modelos Pydantic
class ExtractRequest(BaseModel):
    url: HttpUrl

class VideoRequest(BaseModel):
    image_urls: List[HttpUrl]
    duration_per_image: float = 3.0
    transition_duration: float = 0.5
    resolution: str = "1920x1080"
    fps: int = 24
    add_fade: bool = True

class VideoResponse(BaseModel):
    video_id: str
    status: str
    message: str

# Armazenamento temporário de vídeos gerados
video_storage = {}

@app.post("/api/extract-images")
async def extract_images(request: ExtractRequest):
    """Extrai imagens de uma URL específica"""
    try:
        extractor = ImageExtractor()
        images = extractor.extract_from_url(str(request.url))
        
        # Validar URLs (opcional, pode ser pesado em muitas imagens)
        # valid_images = [img for img in images if extractor.validate_image_url(img)]
        
        return {
            "success": True,
            "count": len(images),
            "images": images[:50]  # Limitar a 50 imagens para não sobrecarregar
        }
    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.post("/api/create-video", response_model=VideoResponse)
async def create_video(request: VideoRequest, background_tasks: BackgroundTasks):
    """Inicia criação do vídeo"""
    video_id = str(uuid.uuid4())
    
    # Parse resolution
    try:
        width, height = map(int, request.resolution.split('x'))
    except:
        width, height = 1920, 1080
    
    # Criar vídeo em background para não travar a requisição
    def generate_video_task():
        try:
            generator = VideoGenerator()
            output_path = generator.create_video(
                image_urls=[str(url) for url in request.image_urls],
                duration_per_image=request.duration_per_image,
                transition_duration=request.transition_duration,
                resolution=(width, height),
                fps=request.fps,
                add_fade=request.add_fade
            )
            
            # Mover para pasta de vídeos gerados
            os.makedirs("generated_videos", exist_ok=True)
            final_path = f"generated_videos/{video_id}.mp4"
            shutil.move(output_path, final_path)
            
            video_storage[video_id] = {
                "path": final_path,
                "status": "completed"
            }
        except Exception as e:
            video_storage[video_id] = {
                "status": "error",
                "message": str(e)
            }
    
    background_tasks.add_task(generate_video_task)
    video_storage[video_id] = {"status": "processing"}
    
    return VideoResponse(
        video_id=video_id,
        status="processing",
        message="Vídeo está sendo gerado"
    )

@app.get("/api/video-status/{video_id}")
async def check_status(video_id: str):
    """Verifica status da geração do vídeo"""
    if video_id not in video_storage:
        raise HTTPException(status_code=404, detail="Vídeo não encontrado")
    
    return video_storage[video_id]

@app.get("/api/download/{video_id}")
async def download_video(video_id: str):
    """Download do vídeo gerado"""
    if video_id not in video_storage or video_storage[video_id].get("status") != "completed":
        raise HTTPException(status_code=404, detail="Vídeo não pronto ou não encontrado")
    
    file_path = video_storage[video_id]["path"]
    if not os.path.exists(file_path):
        raise HTTPException(status_code=404, detail="Arquivo não encontrado")
    
    return FileResponse(
        file_path,
        media_type="video/mp4",
        filename=f"video_gerado_{video_id}.mp4"
    )

@app.post("/api/upload-images")
async def upload_images(files: List[UploadFile] = File(...)):
    """Upload direto de imagens"""
    upload_dir = "uploads"
    os.makedirs(upload_dir, exist_ok=True)
    
    urls = []
    for file in files:
        if file.content_type.startswith("image/"):
            file_path = os.path.join(upload_dir, f"{uuid.uuid4()}_{file.filename}")
            with open(file_path, "wb") as buffer:
                shutil.copyfileobj(file.file, buffer)
            # Em produção, aqui você faria upload para CDN/S3
            urls.append(f"file://{os.path.abspath(file_path)}")
    
    return {"success": True, "urls": urls}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

    <script>
        const API_URL = 'http://localhost:8000';
        let selectedImages = [];
        let availableImages = [];

        function switchTab(tab) {
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
            
            event.target.classList.add('active');
            document.getElementById(tab + '-tab').classList.add('active');
        }

        function showStatus(message, type = 'info') {
            const statusBar = document.getElementById('status-bar');
            statusBar.textContent = message;
            statusBar.className = 'status-bar ' + type;
            statusBar.style.display = 'block';
        }

        function hideStatus() {
            document.getElementById('status-bar').style.display = 'none';
        }

        async function extractImages() {
            const url = document.getElementById('pageUrl').value;
            if (!url) {
                showStatus('Por favor, insira uma URL válida', 'error');
                return;
            }

            const spinner = document.getElementById('extract-spinner');
            spinner.style.display = 'inline-block';

            try {
                const response = await fetch(`${API_URL}/api/extract-images`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ url: url })
                });

                const data = await response.json();
                
                if (data.success) {
                    availableImages = data.images;
                    renderImages();
                    showStatus(`${data.count} imagens encontradas! Selecione as desejadas.`, 'success');
                } else {
                    showStatus('Erro ao extrair imagens', 'error');
                }
            } catch (error) {
                showStatus('Erro de conexão com o servidor', 'error');
            } finally {
                spinner.style.display = 'none';
            }
        }

        function renderImages() {
            const grid = document.getElementById('imagesGrid');
            grid.innerHTML = '';
            
            availableImages.forEach((img, index) => {
                const card = document.createElement('div');
                card.className = 'image-card';
                card.onclick = () => toggleImageSelection(img, card);
                
                card.innerHTML = `
                    <img src="${img}" alt="Imagem ${index + 1}" loading="lazy">
                    <div class="order-badge">${selectedImages.indexOf(img) + 1}</div>
                `;
                
                grid.appendChild(card);
            });

            document.getElementById('images-section').style.display = 'block';
            document.getElementById('settings-section').style.display = 'block';
        }

        function toggleImageSelection(url, card) {
            const index = selectedImages.indexOf(url);
            
            if (index > -1) {
                selectedImages.splice(index, 1);
                card.classList.remove('selected');
            } else {
                selectedImages.push(url);
                card.classList.add('selected');
            }
            
            updateBadges();
        }

        function updateBadges() {
            const cards = document.querySelectorAll('.image-card');
            cards.forEach(card => {
                const badge = card.querySelector('.order-badge');
                const img = card.querySelector('img').src;
                const order = selectedImages.indexOf(img);
                
                if (order > -1) {
                    badge.textContent = order + 1;
                    badge.style.display = 'flex';
                }
            });
        }

        function clearSelection() {
            selectedImages = [];
            document.querySelectorAll('.image-card').forEach(card => {
                card.classList.remove('selected');
            });
        }

        function selectAll() {
            selectedImages = [...availableImages];
            document.querySelectorAll('.image-card').forEach(card => {
                card.classList.add('selected');
            });
            updateBadges();
        }

        // Upload handlers
        function handleDragOver(e) {
            e.preventDefault();
            e.currentTarget.classList.add('dragover');
        }

        function handleDragLeave(e) {
            e.currentTarget.classList.remove('dragover');
        }

        function handleDrop(e) {
            e.preventDefault();
            e.currentTarget.classList.remove('dragover');
            handleFiles(e.dataTransfer.files);
        }

        async function handleFiles(files) {
            const formData = new FormData();
            for (let file of files) {
                if (file.type.startsWith('image/')) {
                    formData.append('files', file);
                }
            }

            showStatus('Fazendo upload das imagens...', 'info');
            
            try {
                const response = await fetch(`${API_URL}/api/upload-images`, {
                    method: 'POST',
                    body: formData
                });
                
                const data = await response.json();
                if (data.success) {
                    availableImages = data.urls;
                    renderImages();
                    showStatus(`${data.urls.length} imagens carregadas!`, 'success');
                }
            } catch (error) {
                showStatus('Erro no upload das imagens', 'error');
            }
        }

        async function generateVideo() {
            if (selectedImages.length < 2) {
                showStatus('Selecione pelo menos 2 imagens', 'error');
                return;
            }

            const spinner = document.getElementById('generate-spinner');
            spinner.style.display = 'inline-block';

            const config = {
                image_urls: selectedImages,
                duration_per_image: parseFloat(document.getElementById('duration').value),
                transition_duration: parseFloat(document.getElementById('transition').value),
                resolution: document.getElementById('resolution').value,
                fps: parseInt(document.getElementById('fps').value),
                add_fade: document.getElementById('addFade').checked
            };

            try {
                const response = await fetch(`${API_URL}/api/create-video`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(config)
                });

                const data = await response.json();
                
                if (data.video_id) {
                    showStatus('Vídeo em processamento... Aguarde', 'info');
                    document.getElementById('progress-bar').style.display = 'block';
                    await checkVideoStatus(data.video_id);
                }
            } catch (error) {
                showStatus('Erro ao iniciar geração do vídeo', 'error');
                spinner.style.display = 'none';
            }
        }

        async function checkVideoStatus(videoId) {
            const checkInterval = setInterval(async () => {
                try {
                    const response = await fetch(`${API_URL}/api/video-status/${videoId}`);
                    const data = await response.json();

                    if (data.status === 'completed') {
                        clearInterval(checkInterval);
                        document.getElementById('progress-fill').style.width = '100%';
                        showVideoResult(videoId);
                        document.getElementById('generate-spinner').style.display = 'none';
                    } else if (data.status === 'error') {
                        clearInterval(checkInterval);
                        showStatus('Erro ao gerar vídeo: ' + data.message, 'error');
                        document.getElementById('generate-spinner').style.display = 'none';
                    } else {
                        // Simular progresso
                        const currentWidth = parseFloat(document.getElementById('progress-fill').style.width) || 0;
                        if (currentWidth < 90) {
                            document.getElementById('progress-fill').style.width = (currentWidth + 10) + '%';
                        }
                    }
                } catch (error) {
                    clearInterval(checkInterval);
                }
            }, 2000);
        }

        function showVideoResult(videoId) {
            const videoUrl = `${API_URL}/api/download/${videoId}`;
            const video = document.getElementById('result-video');
            video.src = videoUrl;
            video.load();
            
            document.getElementById('download-link').href = videoUrl;
            document.getElementById('video-result').style.display = 'block';
            
            // Scroll para o resultado
            document.getElementById('video-result').scrollIntoView({ behavior: 'smooth' });
        }

        function resetApp() {
            selectedImages = [];
            availableImages = [];
            document.getElementById('images-section').style.display = 'none';
            document.getElementById('settings-section').style.display = 'none';
            document.getElementById('video-result').style.display = 'none';
            document.getElementById('imagesGrid').innerHTML = '';
            document.getElementById('pageUrl').value = '';
            document.getElementById('progress-bar').style.display = 'none';
            document.getElementById('progress-fill').style.width = '0%';
            hideStatus();
        }
    </script>
</body>
</html>
```

## 🚀 Como Executar

### 1. Instalar dependências do Backend
```bash
cd backend
pip install -r requirements.txt
```

> **Nota:** O MoviePy requer FFmpeg instalado no sistema. 
> - **Windows:** `choco install ffmpeg` ou baixe em [ffmpeg.org](https://ffmpeg.org/download.html)
> - **Linux:** `sudo apt-get install ffmpeg`
> - **Mac:** `brew install ffmpeg`

### 2. Iniciar o Backend
```bash
python main.py
```
O servidor estará rodando em `http://localhost:8000`

### 3. Abrir o Frontend
Simplesmente abra o arquivo `frontend/index.html` no navegador ou use uma extensão como Live Server no VS Code.

## ✨ Funcionalidades Principais

<details>
<summary><b>📥 Extração de Imagens</b></summary>

- **Web Scraping**: Extrai automaticamente todas as imagens `<img>` de qualquer URL
- **Upload Direto**: Suporte a drag-and-drop de múltiplas imagens locais
- **Validação**: Verifica formato e acessibilidade das imagens
- **Pré-visualização**: Grid visual com seleção múltipla ordenada
</details>

<details>
<summary><b>🎞️ Geração de Vídeo</b></summary>

- **Codecs**: H.264 (compatível universal) via FFmpeg
- **Resoluções**: Full HD, 4K, Stories (9:16), HD
- **Transições**: Fade in/out entre imagens
- **Ajuste Inteligente**: Mantém aspect ratio com letterbox automático
- **Processamento**: Assíncrono (não trava a interface)
</details>

<details>
<summary><b>⚙️ Configurações Avançadas</b></summary>

- **Duração**: Segundos por imagem (0.5 até 10s)
- **FPS**: 24 (cinema), 30 (padrão), 60 (fluído)
- **Ordem Personalizada**: Clique nas imagens na sequência desejada
- **Compactação**: Otimização automática de bitrate
</details>
